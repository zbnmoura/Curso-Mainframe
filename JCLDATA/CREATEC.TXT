-- CRIACAO DA TABELA TBCLIENTE

CREATE TABLE TBCLIENTE
(CODCLI   CHAR    (04) NOT NULL,
 NOMECLI  VARCHAR (30) NOT NULL,
 ENDCLI   VARCHAR (40)         ,
 FONECLI  CHAR    (09)         ,
 EMAILCLI VARCHAR (50)         ,
 CPFCLI   CHAR    (11)         ,
 SITUACAO CHAR    (01)         ,
 PRIMARY KEY(CODCLI)
 ) ;

-- CRIACAO DO INDICE IDXCLIENTE

CREATE UNIQUE INDEX IDXCLIENTE
ON TBCLIENTE (CODCLI)

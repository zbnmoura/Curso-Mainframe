CREATE TABLE TBPRODUTO
( CODPROD            CHAR(4) UNIQUE NOT NULL,
  DESCPROD           VARCHAR(20) NOT NULL,
  UNIDPROD           CHAR(5),
  LOCALPROD          CHAR(5),
  QTDEST             DECIMAL(5,0),
  QTDMAX             DECIMAL(5,0),
  QTDMIN             DECIMAL(5,0),
  PRECOCOMPRA        DECIMAL(8,2) CHECK (PRECOCOMPRA >=0),
  PRECOVENDA         DECIMAL(8,2) CHECK (PRECOVENDA >=0),
  PERCOMIS           DECIMAL(4,2) CHECK(PERCOMIS >=0),
  CONSTRAINT QTD01 CHECK (QTDMAX > QTDMIN),
  CONSTRAINT QTD02 CHECK (QTDMIN < QTDMAX)
);

-- CRIACAO DO INDICE DA TABELA TBPRODUTO

CREATE UNIQUE INDEX IDXPRODUTO
ON TBPRODUTO (CODPROD);

-- CRIACAO DA CHAVE PRIMARIA DA

ALTER TABLE TBPRODUTO ADD PRIMARY KEY(CODPROD)
